# Recommended WSL2 Configuration
# Location: C:\Users\<YourUsername>\.wslconfig
# Create this file if it doesn't exist.

[wsl2]
# Increase memory allocation if you have more physical RAM.
# 15Gi is often the default (50% or 80% of host RAM).
# If you have 32GB RAM, set this to 24GB or more.
memory=24GB 

# Increase swap file size to prevent OOM (Out Of Memory) crashes during compilation.
# 16GB is a safe value for building heavy CUDA extensions.
swap=16GB

# Use all available processors
processors=16

# This will take effect after running 'wsl --shutdown' in PowerShell.
